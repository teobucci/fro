\begin{equation*}
	y_{j} \geq 3\ \ \overset{\text{(A)}}{\Rightarrow } \ \ \varphi =1\ \ \overset{\text{(B)}}{\Rightarrow } \ \ z_{j} \geq 2
\end{equation*}
